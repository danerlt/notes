# 主流模型对比

## 概念介绍

2024-2025 年，AI 大模型进入群雄争霸时代。闭源模型（GPT-4o、Claude 3.x、Gemini 1.5/2.0）持续保持领先，开源模型（LLaMA-3、Qwen2.5、DeepSeek-V3）快速追赶，能力差距大幅缩小。本文梳理主流 LLM 的核心特性，帮助根据需求选择合适的模型。

## 代码示例（Python）

### 统一 API 调用封装

```python
from openai import OpenAI
from anthropic import Anthropic
from typing import Optional


class UnifiedLLMClient:
    """统一的 LLM 调用封装"""

    def __init__(self):
        self.openai_client = OpenAI()
        self.anthropic_client = Anthropic()

    def chat(
        self,
        provider: str,
        model: str,
        messages: list,
        temperature: float = 0.7,
        max_tokens: int = 1024
    ) -> str:
        if provider == "openai":
            response = self.openai_client.chat.completions.create(
                model=model,
                messages=messages,
                temperature=temperature,
                max_tokens=max_tokens
            )
            return response.choices[0].message.content

        elif provider == "anthropic":
            # 提取 system message
            system = ""
            user_messages = []
            for msg in messages:
                if msg["role"] == "system":
                    system = msg["content"]
                else:
                    user_messages.append(msg)

            response = self.anthropic_client.messages.create(
                model=model,
                system=system,
                messages=user_messages,
                temperature=temperature,
                max_tokens=max_tokens
            )
            return response.content[0].text

        elif provider == "deepseek":
            # DeepSeek 兼容 OpenAI API
            client = OpenAI(
                api_key="your-deepseek-key",
                base_url="https://api.deepseek.com"
            )
            response = client.chat.completions.create(
                model=model,
                messages=messages,
                temperature=temperature,
                max_tokens=max_tokens
            )
            return response.choices[0].message.content

        else:
            raise ValueError(f"不支持的 provider: {provider}")


# 使用示例
client = UnifiedLLMClient()
messages = [{"role": "user", "content": "用一句话解释什么是 Transformer"}]

# 对比不同模型的回答
models = [
    ("openai", "gpt-4o"),
    ("anthropic", "claude-3-5-sonnet-20241022"),
    ("deepseek", "deepseek-chat"),
]

for provider, model in models:
    try:
        response = client.chat(provider, model, messages)
        print(f"\n[{model}]:\n{response}")
    except Exception as e:
        print(f"[{model}] 错误: {e}")
```

### 使用 LiteLLM 统一调用（2024 年推荐）

```python
# pip install litellm
import litellm

# LiteLLM 支持 100+ 模型，统一 OpenAI 格式
models_to_test = [
    "gpt-4o",
    "claude-3-5-sonnet-20241022",
    "gemini/gemini-2.0-flash",
    "deepseek/deepseek-chat",
    "ollama/qwen2.5:7b",  # 本地模型
]

messages = [{"role": "user", "content": "Hello! 介绍一下你自己。"}]

for model in models_to_test:
    try:
        response = litellm.completion(
            model=model,
            messages=messages,
            max_tokens=200,
            temperature=0.7
        )
        print(f"\n[{model}]:")
        print(response.choices[0].message.content)
        print(f"Cost: ${response._hidden_params.get('response_cost', 'N/A')}")
    except Exception as e:
        print(f"[{model}] Error: {e}")
```

### 模型选择决策工具

```python
def recommend_model(
    task: str,
    budget: str = "medium",  # "low", "medium", "high"
    requires_chinese: bool = False,
    requires_open_source: bool = False,
    requires_long_context: bool = False,
    requires_reasoning: bool = False,
) -> dict:
    """根据需求推荐模型"""

    recommendations = []

    if requires_open_source:
        if requires_chinese:
            recommendations.append({
                "model": "Qwen2.5-72B-Instruct",
                "provider": "阿里云/自部署",
                "reason": "中文最强开源模型"
            })
            recommendations.append({
                "model": "DeepSeek-V3",
                "provider": "DeepSeek API/自部署",
                "reason": "综合能力强，API 价格极低"
            })
        elif requires_reasoning:
            recommendations.append({
                "model": "DeepSeek-R1",
                "provider": "DeepSeek API/自部署",
                "reason": "开源推理模型标杆"
            })
        else:
            recommendations.append({
                "model": "LLaMA-3.3-70B",
                "provider": "自部署",
                "reason": "英文最强开源模型之一"
            })
    else:
        if requires_reasoning:
            if budget == "high":
                recommendations.append({"model": "o3", "provider": "OpenAI", "reason": "最强推理"})
            else:
                recommendations.append({"model": "o3-mini", "provider": "OpenAI", "reason": "推理+低成本"})
        elif requires_long_context:
            recommendations.append({"model": "gemini-2.0-pro", "provider": "Google", "reason": "1M 上下文"})
            recommendations.append({"model": "claude-3-5-sonnet-20241022", "provider": "Anthropic", "reason": "200K 上下文，精准遵循指令"})
        elif budget == "low":
            recommendations.append({"model": "gpt-4o-mini", "provider": "OpenAI", "reason": "低成本高性能"})
            recommendations.append({"model": "claude-3-haiku", "provider": "Anthropic", "reason": "最快最便宜"})
        else:
            recommendations.append({"model": "gpt-4o", "provider": "OpenAI", "reason": "综合最强，生态最好"})
            recommendations.append({"model": "claude-3-5-sonnet-20241022", "provider": "Anthropic", "reason": "代码能力顶级"})

    return {"task": task, "recommendations": recommendations}


# 示例
result = recommend_model(
    task="中文客服机器人",
    budget="medium",
    requires_chinese=True,
    requires_open_source=True
)
print(result)
```

## 小结

- **2025 年格局**：闭源模型（OpenAI o3/GPT-4o、Claude 3.7）保持领先，开源模型（DeepSeek-R1、Qwen2.5）大幅缩小差距
- **推理模型**是 2025 年最重要的趋势，通过扩展测试时计算解决复杂推理任务
- **中文场景**优先选择 Qwen2.5 或 DeepSeek 系列，性价比极高
- **私有化部署**推荐 Qwen2.5-7B/14B（消费级 GPU 可运行）或 DeepSeek-R1-Distill-7B
- **成本控制**：使用 LiteLLM 统一管理多模型调用，根据任务难度路由到合适的模型
